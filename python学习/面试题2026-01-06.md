# 示例1：原列表每个数乘2 
a = [1,2,3,4] res = [i * 2 for i in a] 
print(res) 
# 输出：[2,4,6,8] 
# 示例2：原字符串列表转大写 
words = ["apple", "banana"] 
res = [word.upper() for word in words] 
print(res) 
# 输出：["APPLE", "BANANA"]

# 示例1：筛选原列表中的偶数 
a = [1,2,3,4,5,6] 
res = [i for i in a if i % 2 == 0] 
print(res) # 输出：[2,4,6]

words = ["a", "bb", "ccc", "dddd"] 
res = [word for word in words if len(word) > 3] 
print(res) 
# 输出：["dddd"]

# 示例1：筛选原列表中的奇数，再乘3 
a = [1,2,3,4,5]

# 示例2：筛选原列表中以"a"开头的字符串，再转小写

print(res) 
# 输出：["apple", "ant"]

13.一行代码写出1+2+3+10248

num1 = reduce(lambda x,y :x+y,[1,2,3,10248])

from functools import reduce nums = [1,2,3,4] 
# 累积函数：x是前n个元素的和，y是当前元素，返回x+y 
total = reduce(lambda x, y: x + y, nums) print(total) 
# 输出：10（1+2=3 → 3+3=6 → 6+4=10）
