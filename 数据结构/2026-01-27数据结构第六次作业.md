```python
# 树结点类 
class TreeNode: 
	def __init__(self, val=0, left=None, right=None): 
		self.val = val # 设置属性存储左、右子节点的链接 
		self.left = left 
		self.right = right # 输入树的根结点，计算树最大深度 
	
def maxDepth(root): 
	if root is None: 
		return 0 
	# 递归计算左、右子树的深度 
	left_depth = maxDepth(root.left) 
	right_depth = maxDepth(root.right) 
	# 当前节点的深度 = 左右子树深度的最大值 + 1 
	return max(left_depth, right_depth) + 1 
		
# 实例化结点，并构造为树 
n1 = TreeNode('n1') 
n2 = TreeNode('n2') 
n3 = TreeNode('n3',n1,n2) 
n4 = TreeNode('n4') 
n5 = TreeNode('n5',None,n4) 
n6 = TreeNode('n6', n3, n5) 
root = TreeNode('root',n6, None) 
	
# 计算该树的最大深度 
print(maxDepth(root))

```
运行结果：
`4`
