```python
# 结点类
class ListNode:
	def __init__(self, val, next_ = None):
	# 设置属性存储结点的值和后继链接
		self.val = val
		self.next = next_	

class LList:
	def __init__(self):
		self._head = None

# 在位置i处添加结点的方法
	def insert(self, i, val):
		q = ListNode(val)
		if i == 0:
			# 链接结点q与链表首部
			q.next = self._head
			self._head = q
			return
		p = self._head
		count = 1
		while count < i:
			# 移动探针p
			if p is None:
				raise IndexError('插入位置超出链表长度')
			p = p.next	
			count += 1
		# 将结点q链接到合适的结点
		q.next = p.next
		p.next = q

	# 打印链表所有节点的值
	def report(self):
		p = self._head
		result = [ ]
		while p is not None:
		# 移动探针p并将值记录到result列表
			result.append(str(p.val))
			p = p.next
		print(result) 		
```
实例：
```python
mlist = LList()
nums = list(range(20))

for i in range(19, 0, -3):
	mlist.insert(0, nums[i])

for i in range(5):
	mlist.insert(2*i, i**2)

mlist.report()
```
