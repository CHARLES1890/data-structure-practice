```python
def findPeakElement(nums):
    left, right = 0, len(nums) - 1
    while left < right:
        mid = (left + right) // 2
        # 峰值不出现在数组边缘，因此 mid 不会是首尾
        if nums[mid] > nums[mid + 1]:
            # 说明峰值在左侧
            right = mid
        else:
            # 说明峰值在右侧
            left = mid + 1
    return left
```
