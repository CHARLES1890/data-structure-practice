def uniquePathsWithObstacles(obstacleGrid):
    m = len(obstacleGrid)
    n = len(obstacleGrid[0])
    
    # 设置 dp 数组初始值
    dp = [[0 for _ in range(n)] for _ in range(m)]
    
    # 初始化第一列
    for i in range(m):
        if obstacleGrid[i][0] == 1:
            break
        dp[i][0] = 1
    
    # 初始化第一行
    for j in range(n):
        if obstacleGrid[0][j] == 1:
            break
        dp[0][j] = 1
    
    # 填充 dp 表
    for i in range(1, m):
        for j in range(1, n):
            # 若遇到障碍物
            if obstacleGrid[i][j] == 1:
                continue
            # 状态转移方程
            dp[i][j] = dp[i-1][j] + dp[i][j-1]
    
    # 返回结果
    return dp[m-1][n-1]
