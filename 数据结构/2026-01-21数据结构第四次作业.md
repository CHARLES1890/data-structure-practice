from collections import deque 
def longestSubarray(nums, limit): 
	ans = 0 
	left, right = 0, 0 
	maxq = deque() # 存放滑动窗口最大值坐标 
	minq = deque() # 存放滑动窗口最小值坐标 
	
	for right in range(len(nums)): 
	# 若 nums[right] 大于等于 maxq 队尾元素，更新 maxq 
		while maxq and nums[right] >= nums[maxq[-1]]: 
			maxq.pop() 
		maxq.append(right) 
		
		# 若 nums[right] 小于等于 minq 队尾元素，更新 minq 
		while minq and nums[right] <= nums[minq[-1]]: 
			minq.pop() 
		minq.append(right)
		
		 # 若当前滑动窗口的最大绝对差超过了 limit 
		 while left <= right and nums[maxq[0]] - nums[minq[0]] > limit: 
		 # 窗口向右移，并检查 maxq、minq 队首元素是否已不在窗口内 
			left = left + 1 
			if maxq[0] < left: 
				maxq.popleft() 
			if minq[0] < left: 
				minq.popleft() 
		 
		# 更新 ans，记录满足要求的子数组最大长度 
		ans = max(ans, right - left + 1) 
	return ans
